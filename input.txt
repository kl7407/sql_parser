    id int,
    name char(20),
    primary key (id)
);
create table b (
    id int,
    a_id int,
    primary key (id),
    foreign key (a_id) references a(id)
);
create table c (
    id int,
    a_id int,
    b_id int,
    primary key (id, a_id, b_id),
    foreign key (a_id) references a(id),
    foreign key (b_id) references b(id)
);
create table d (
    id int,
    name char(20) not null,
    a_id int,
    primary key (id),
    foreign key (a_id) references a(id)
);

    primary key (id),

create table person (
    person_id int not null,
    person_name char(20),

    account_number int not null,
    branch_name char(15),
    person_id int,
    assets int,
    primary key(account_number),
    foreign key(person_id) references person(person_id)
);
