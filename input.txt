create table person (
    person_id int not null,
    person_name char(20),
    sex char(1),
    primary key(person_id)
);

create table account (
    account_number int not null,
    branch_name char(15),
    person_name char(20),
    assets int,
    primary key(account_number),
    foreign key(person_name) references person(person_name)
);
